95  if constexpr(!Const) {

101  auto * const direct = value.try_cast<data_type>();
102

106  }

108  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>;
109  static_assert(std::is_invocable_v<decltype(Data), Type *>);
110  auto * const clazz = instance.try_cast<Type>();
111 
112  if constexpr(std::is_array_v<data_type>) {
113  using underlying_type = std::remove_extent_t<data_type>;
114  auto * const direct = value.try_cast<underlying_type>();
115  auto * const idx = index.try_cast<std::size_t>();
116 
117  if(clazz && idx && (direct || value.convert<underlying_type>())) {
118  std::invoke(Data, clazz)[*idx] = direct ? *direct : value.cast<underlying_type>();
119  accepted = true;
120  }
121  } else {
122  auto * const direct = value.try_cast<data_type>();
123

125  std::invoke(Data, clazz) = (direct ? *direct : value.cast<data_type>());
126  accepted = true;
127  }
128  }
129  } else {

157 template<typename Type, auto Data, typename Policy>
158 meta_any getter([[maybe_unused]] meta_any instance, [[maybe_unused]] meta_any index) {
159  auto dispatch = [](auto &&value) {
160  if constexpr(std::is_same_v<Policy, as_void_t>) {
161  return meta_any{std::in_place_type<void>, std::forward<decltype(value)>(value)};
162  } else if constexpr(std::is_same_v<Policy, as_ref_t>) {
163  return meta_any{std::ref(std::forward<decltype(value)>(value))};
164  } else {
165  static_assert(std::is_same_v<Policy, as_is_t>);
166  return meta_any{std::forward<decltype(value)>(value)};
167  }
168  };
169 
170  if constexpr(std::is_function_v<std::remove_reference_t<std::remove_pointer_t<decltype(Data)>>> || std::is_member_function_pointer_v<decltype(Data)>) {
171  static_assert(std::is_invocable_v<decltype(Data), Type &>);
172  auto * const clazz = instance.try_cast<Type>();
173  return clazz ? dispatch(std::invoke(Data, *clazz)) : meta_any{};
174  } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
175  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>;
176  static_assert(std::is_invocable_v<decltype(Data), Type *>);
177  auto * const clazz = instance.try_cast<Type>();
178 
179  if constexpr(std::is_array_v<data_type>) {
180  auto * const idx = index.try_cast<std::size_t>();
181  return (clazz && idx) ? dispatch(std::invoke(Data, clazz)[*idx]) : meta_any{};
182  } else {
183  return clazz ? dispatch(std::invoke(Data, clazz)) : meta_any{};
184  }
185  } else {
186  static_assert(std::is_pointer_v<std::decay_t<decltype(Data)>>);
187 
188  if constexpr(std::is_array_v<std::remove_pointer_t<decltype(Data)>>) {
189  auto * const idx = index.try_cast<std::size_t>();
190  return idx ? dispatch((*Data)[*idx]) : meta_any{};
191  } else {
192  return dispatch(*Data);
193  }
194  }
195 }
