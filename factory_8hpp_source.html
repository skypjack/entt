121 template<typename Type, auto Data, typename Policy>
122 [[nodiscard]] meta_any getter([[maybe_unused]] meta_handle instance) {
123  [[maybe_unused]] auto dispatch = [](auto &&value) {
124  if constexpr(std::is_same_v<Policy, as_void_t>) {
125  return meta_any{std::in_place_type<void>, std::forward<decltype(value)>(value)};
126  } else if constexpr(std::is_same_v<Policy, as_ref_t>) {
127  return meta_any{std::ref(std::forward<decltype(value)>(value))};
128  } else if constexpr(std::is_same_v<Policy, as_cref_t>) {
129  return meta_any{std::cref(std::forward<decltype(value)>(value))};
130  } else {
131  static_assert(std::is_same_v<Policy, as_is_t>, "Policy not supported");
132  return meta_any{std::forward<decltype(value)>(value)};
133  }
134  };
135 
136  if constexpr(std::is_function_v<std::remove_reference_t<std::remove_pointer_t<decltype(Data)>>> || std::is_member_function_pointer_v<decltype(Data)>) {
137  auto * const clazz = instance->try_cast<std::conditional_t<std::is_invocable_v<decltype(Data), const Type *>, const Type, Type>>();
138  return clazz ? dispatch(std::invoke(Data, *clazz)) : meta_any{};
139  } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
140  if constexpr(std::is_array_v<std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>>) {
141  return meta_any{};
142  } else {
143  auto * const clazz = instance->try_cast<std::conditional_t<std::is_same_v<Policy, as_ref_t>, Type, const Type>>();
144  return clazz ? dispatch(std::invoke(Data, clazz)) : meta_any{};
145  }
146  } else if constexpr(std::is_pointer_v<std::decay_t<decltype(Data)>>) {
147  if constexpr(std::is_array_v<std::remove_pointer_t<decltype(Data)>>) {
148  return meta_any{};
149  } else {
150  return dispatch(*Data);
151  }
152  } else {
153  return dispatch(Data);
154  }
155 }
