146[[nodiscard]] inline const void *try_cast(const meta_context &context, const meta_type_node &from, const meta_type_node &to, const void *instance) noexcept {
147 if(from.info && to.info && *from.info == *to.info) {
148 return instance;
149 }
150
151 if(from.details) {
152 for(auto &&curr: from.details->base) {
153 if(const void *elem = try_cast(context, curr.second.type(context), to, curr.second.cast(instance)); elem) {
154 return elem;
155 }
156 }
157 }
158
159 return nullptr;

167template<typename Type>
168[[nodiscard]] meta_type_node resolve(const meta_context &context) noexcept {
169 static_assert(std::is_same_v<Type, std::remove_const_t<std::remove_reference_t<Type>>>, "Invalid type");
170
171 if(auto *elem = try_resolve(context, type_id<Type>()); elem) {
172 return *elem;
173 }
174
175 meta_type_node node{
176 &type_id<Type>(),
177 type_id<Type>().hash(),
178 (std::is_arithmetic_v<Type> ? meta_traits::is_arithmetic : meta_traits::is_none)
179 | (std::is_integral_v<Type> ? meta_traits::is_integral : meta_traits::is_none)
180 | (std::is_signed_v<Type> ? meta_traits::is_signed : meta_traits::is_none)
181 | (std::is_array_v<Type> ? meta_traits::is_array : meta_traits::is_none)
182 | (std::is_enum_v<Type> ? meta_traits::is_enum : meta_traits::is_none)
183 | (std::is_class_v<Type> ? meta_traits::is_class : meta_traits::is_none)
184 | (is_meta_pointer_like_v<Type> ? meta_traits::is_meta_pointer_like : meta_traits::is_none)
185 | (is_complete_v<meta_sequence_container_traits<Type>> ? meta_traits::is_meta_sequence_container : meta_traits::is_none)
186 | (is_complete_v<meta_associative_container_traits<Type>> ? meta_traits::is_meta_associative_container : meta_traits::is_none),
187 size_of_v<Type>,
188 &resolve<Type>,
189 &resolve<std::remove_cv_t<std::remove_pointer_t<Type>>>};
190
191 if constexpr(std::is_default_constructible_v<Type>) {
192 node.default_constructor = +[](const meta_ctx &ctx) {
193 return meta_any{ctx, std::in_place_type<Type>};
194 };
195 }
196
197 if constexpr(std::is_arithmetic_v<Type>) {
198 node.conversion_helper = +[](void *bin, const void *value) {
199 return bin ? static_cast<double>(*static_cast<Type *>(bin) = static_cast<Type>(*static_cast<const double *>(value))) : static_cast<double>(*static_cast<const Type *>(value));
200 };
201 } else if constexpr(std::is_enum_v<Type>) {
202 node.conversion_helper = +[](void *bin, const void *value) {
203 return bin ? static_cast<double>(*static_cast<Type *>(bin) = static_cast<Type>(static_cast<std::underlying_type_t<Type>>(*static_cast<const double *>(value)))) : static_cast<double>(*static_cast<const Type *>(value));
204 };
205 }
206
207 if constexpr(!std::is_same_v<Type, void> && !std::is_function_v<Type>) {
208 node.from_void = +[](const meta_ctx &ctx, void *element, const void *as_const) {
209 if(element) {
210 return meta_any{ctx, std::in_place_type<std::decay_t<Type> &>, *static_cast<std::decay_t<Type> *>(element)};
211 }
212
213 return meta_any{ctx, std::in_place_type<const std::decay_t<Type> &>, *static_cast<const std::decay_t<Type> *>(as_const)};
214 };
215 }
216
217 if constexpr(is_complete_v<meta_template_traits<Type>>) {
218 node.templ = meta_template_node{
219 meta_template_traits<Type>::args_type::size,
220 &resolve<typename meta_template_traits<Type>::class_type>,
221 +[](const meta_context &area, const std::size_t index) noexcept { return meta_arg_node(area, typename meta_template_traits<Type>::args_type{}, index); }};
222 }
223
224 return node;
225}
