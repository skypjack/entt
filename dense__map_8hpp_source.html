281 for(auto it = begin(bucket), last = end(bucket); it != last; ++it) {

292 for(auto it = cbegin(bucket), last = cend(bucket); it != last; ++it) {
293 if(packed.second()(it->first, key)) {

301 template<typename Other, typename... Args>
302 [[nodiscard]] auto insert_or_do_nothing(Other &&key, Args &&...args) {
303 const auto index = key_to_bucket(key);
304
305 if(auto it = constrained_find(key, index); it != end()) {
306 return std::make_pair(it, false);
307 }
308
309 packed.first().emplace_back(sparse.first()[index], std::piecewise_construct, std::forward_as_tuple(std::forward<Other>(key)), std::forward_as_tuple(std::forward<Args>(args)...));
310 sparse.first()[index] = packed.first().size() - 1u;
311 rehash_if_required();
312
313 return std::make_pair(--end(), true);
314 }
315
316 template<typename Other, typename Arg>
317 [[nodiscard]] auto insert_or_overwrite(Other &&key, Arg &&value) {
318 const auto index = key_to_bucket(key);
319
320 if(auto it = constrained_find(key, index); it != end()) {
321 it->second = std::forward<Arg>(value);
322 return std::make_pair(it, false);
323 }
324
325 packed.first().emplace_back(sparse.first()[index], std::forward<Other>(key), std::forward<Arg>(value));
326 sparse.first()[index] = packed.first().size() - 1u;
327 rehash_if_required();
328
329 return std::make_pair(--end(), true);
330 }

332 void move_and_pop(const std::size_t pos) {
333 if(const auto last = size() - 1u; pos != last) {
334 size_type *curr = sparse.first().data() + key_to_bucket(packed.first().back().element.first);
335 packed.first()[pos] = std::move(packed.first().back());
336 for(; *curr != last; curr = &packed.first()[*curr].next) {}
337 *curr = pos;
338 }
339
340 packed.first().pop_back();
341 }

343 void rehash_if_required() {
344 if(size() > (bucket_count() * max_load_factor())) {
345 rehash(bucket_count() * 2u);
346 }
347 }

613 template<typename... Args>
614 std::pair<iterator, bool> emplace([[maybe_unused]] Args &&...args) {
615 if constexpr(sizeof...(Args) == 0u) {
616 return insert_or_do_nothing(key_type{});
617 } else if constexpr(sizeof...(Args) == 1u) {
618 return insert_or_do_nothing(std::forward<Args>(args).first..., std::forward<Args>(args).second...);
619 } else if constexpr(sizeof...(Args) == 2u) {
620 return insert_or_do_nothing(std::forward<Args>(args)...);
621 } else {
622 auto &node = packed.first().emplace_back(packed.first().size(), std::forward<Args>(args)...);
623 const auto index = key_to_bucket(node.element.first);
624
625 if(auto it = constrained_find(node.element.first, index); it != end()) {
626 packed.first().pop_back();
627 return std::make_pair(it, false);
628 }
629
630 std::swap(node.next, sparse.first()[index]);
631 rehash_if_required();
632
633 return std::make_pair(--end(), true);
634 }
635 }

947 }

955 return key_to_bucket(key);
956 }

963 return size() / static_cast<float>(bucket_count());
964 }

971 return threshold;
972 }

978 void max_load_factor(const float value) {
979 ENTT_ASSERT(value > 0.f, "Invalid load factor");
980 threshold = value;
981 rehash(0u);
982 }

989 void rehash(const size_type cnt) {
990 auto value = cnt > minimum_capacity ? cnt : minimum_capacity;
991 const auto cap = static_cast<size_type>(size() / max_load_factor());
992 value = value > cap ? value : cap;
993
994 if(const auto sz = next_power_of_two(value); sz != bucket_count()) {
995 sparse.first().resize(sz);
996
997 for(auto &&elem: sparse.first()) {
998 elem = std::numeric_limits<size_type>::max();
999 }
1000
1001 for(size_type pos{}, last = size(); pos < last; ++pos) {
1002 const auto index = key_to_bucket(packed.first()[pos].element.first);
1003 packed.first()[pos].next = std::exchange(sparse.first()[index], pos);
1004 }
1005 }
1006 }
1007
1013 void reserve(const size_type cnt) {
1014 packed.first().reserve(cnt);
1015 rehash(static_cast<size_type>(std::ceil(cnt / max_load_factor())));
1016 }

1022 [[nodiscard]] hasher hash_function() const {
1023 return sparse.second();
1024 }

1030 [[nodiscard]] key_equal key_eq() const {
1031 return packed.second();
1032 }

1037 float threshold;
