140 template<typename... Type>
141 static std::size_t fill_dependencies(type_list<Type...>, [[maybe_unused]] const type_info **buffer, [[maybe_unused]] const std::size_t count) {
142 if constexpr(sizeof...(Type) == 0u) {
143 return {};
144 } else {
145 const type_info *info[sizeof...(Type)]{&type_id<Type>()...};
146 const auto length = count < sizeof...(Type) ? count : sizeof...(Type);
147
148 for(std::size_t pos{}; pos < length; ++pos) {
149 buffer[pos] = info[pos];
150 }
151
152 return length;
153 }
154 }

156 template<typename... RO, typename... RW>
157 void track_dependencies(std::size_t index, const bool requires_registry, type_list<RO...>, type_list<RW...>) {
158 builder.bind(static_cast<id_type>(index));
159 builder.set(type_hash<Registry>::value(), requires_registry || (sizeof...(RO) + sizeof...(RW) == 0u));
160 (builder.ro(type_hash<RO>::value()), ...);
161 (builder.rw(type_hash<RW>::value()), ...);
162 }

380 std::vector<vertex> graph() {
381 std::vector<vertex> adjacency_list{};
382 adjacency_list.reserve(vertices.size());
383 auto adjacency_matrix = builder.graph();
384
385 for(auto curr: adjacency_matrix.vertices()) {
386 const auto iterable = adjacency_matrix.in_edges(curr);
387 std::vector<std::size_t> reachable{};
388
389 for(auto &&edge: adjacency_matrix.out_edges(curr)) {
390 reachable.push_back(edge.second);
391 }
392
393 adjacency_list.emplace_back(iterable.cbegin() == iterable.cend(), vertices[curr], std::move(reachable));
394 }
395
396 return adjacency_list;
397 }
