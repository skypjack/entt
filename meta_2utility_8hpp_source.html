164template<typename Policy = as_is_t, typename Type>
165std::enable_if_t<is_meta_policy_v<Policy>, meta_any> meta_dispatch(const meta_ctx &ctx, [[maybe_unused]] Type &&value) {
166 if constexpr(std::is_same_v<Policy, as_void_t>) {
167 return meta_any{ctx, std::in_place_type<void>};
168 } else if constexpr(std::is_same_v<Policy, as_ref_t>) {
169 return meta_any{ctx, std::in_place_type<Type>, value};
170 } else if constexpr(std::is_same_v<Policy, as_cref_t>) {
171 static_assert(std::is_lvalue_reference_v<Type>, "Invalid type");
172 return meta_any{ctx, std::in_place_type<const std::remove_reference_t<Type> &>, std::as_const(value)};
173 } else {
174 return meta_any{ctx, std::forward<Type>(value)};
175 }
176}

222template<typename Type, auto Data>
223[[nodiscard]] bool meta_setter([[maybe_unused]] meta_handle instance, [[maybe_unused]] meta_any value) {
224 if constexpr(!std::is_same_v<decltype(Data), Type> && !std::is_same_v<decltype(Data), std::nullptr_t>) {
225 if constexpr(std::is_member_function_pointer_v<decltype(Data)> || std::is_function_v<std::remove_reference_t<std::remove_pointer_t<decltype(Data)>>>) {
226 using descriptor = meta_function_helper_t<Type, decltype(Data)>;
227 using data_type = type_list_element_t<descriptor::is_static, typename descriptor::args_type>;
228
229 if(auto *const clazz = instance->try_cast<Type>(); clazz && value.allow_cast<data_type>()) {
230 std::invoke(Data, *clazz, value.cast<data_type>());
231 return true;
232 }
233 } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
234 using data_type = std::remove_reference_t<typename meta_function_helper_t<Type, decltype(Data)>::return_type>;
235
236 if constexpr(!std::is_array_v<data_type> && !std::is_const_v<data_type>) {
237 if(auto *const clazz = instance->try_cast<Type>(); clazz && value.allow_cast<data_type>()) {
238 std::invoke(Data, *clazz) = value.cast<data_type>();
239 return true;
240 }
241 }
242 } else {
243 using data_type = std::remove_reference_t<decltype(*Data)>;
244
245 if constexpr(!std::is_array_v<data_type> && !std::is_const_v<data_type>) {
246 if(value.allow_cast<data_type>()) {
247 *Data = value.cast<data_type>();
248 return true;
249 }
250 }
251 }
252 }
253
254 return false;
255}

271template<typename Type, auto Data, typename Policy = as_is_t>
272[[nodiscard]] std::enable_if_t<is_meta_policy_v<Policy>, meta_any> meta_getter(const meta_ctx &ctx, [[maybe_unused]] meta_handle instance) {
273 if constexpr(std::is_member_pointer_v<decltype(Data)> || std::is_function_v<std::remove_reference_t<std::remove_pointer_t<decltype(Data)>>>) {
274 if constexpr(!std::is_array_v<std::remove_cv_t<std::remove_reference_t<std::invoke_result_t<decltype(Data), Type &>>>>) {
275 if constexpr(std::is_invocable_v<decltype(Data), Type &>) {
276 if(auto *clazz = instance->try_cast<Type>(); clazz) {
277 return meta_dispatch<Policy>(ctx, std::invoke(Data, *clazz));
278 }
279 }
280
281 if constexpr(std::is_invocable_v<decltype(Data), const Type &>) {
282 if(auto *fallback = instance->try_cast<const Type>(); fallback) {
283 return meta_dispatch<Policy>(ctx, std::invoke(Data, *fallback));
284 }
285 }
286 }
287
288 return meta_any{meta_ctx_arg, ctx};
289 } else if constexpr(std::is_pointer_v<decltype(Data)>) {
290 if constexpr(std::is_array_v<std::remove_pointer_t<decltype(Data)>>) {
291 return meta_any{meta_ctx_arg, ctx};
292 } else {
293 return meta_dispatch<Policy>(ctx, *Data);
294 }
295 } else {
296 return meta_dispatch<Policy>(ctx, Data);
297 }
298}
