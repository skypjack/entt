167 void assign(Archive &archive) const {
168 typename entity_traits::entity_type length{};
169 entity_type entt;
170
171 archive(length);
172
173 if constexpr(ignore_as_empty_v<Component>) {
174 while(length--) {
175 archive(entt);
176 const auto entity = reg->valid(entt) ? entt : reg->create(entt);
177 ENTT_ASSERT(entity == entt, "Entity not available for use");
178 reg->template emplace<Component>(entt);
179 }
180 } else {
181 Component instance;
182
183 while(length--) {
184 archive(entt, instance);
185 const auto entity = reg->valid(entt) ? entt : reg->create(entt);
186 ENTT_ASSERT(entity == entt, "Entity not available for use");
187 reg->template emplace<Component>(entt, std::move(instance));
188 }
189 }

303 void destroy(typename Registry::entity_type entt) {
304 if(const auto it = remloc.find(entt); it == remloc.cend()) {
305 const auto local = reg->create();

307 reg->destroy(local);

311 void restore(typename Registry::entity_type entt) {
312 const auto it = remloc.find(entt);
313
314 if(it == remloc.cend()) {
315 const auto local = reg->create();
316 remloc.emplace(entt, std::make_pair(local, true));
317 } else {
318 if(!reg->valid(remloc[entt].first)) {
319 remloc[entt].first = reg->create();
320 }

327 template<typename Container>
328 auto update(int, Container &container) -> decltype(typename Container::mapped_type{}, void()) {
329 // map like container
330 Container other;
331
332 for(auto &&pair: container) {
333 using first_type = std::remove_const_t<typename std::decay_t<decltype(pair)>::first_type>;
334 using second_type = typename std::decay_t<decltype(pair)>::second_type;
335
336 if constexpr(std::is_same_v<first_type, entity_type> && std::is_same_v<second_type, entity_type>) {
337 other.emplace(map(pair.first), map(pair.second));
338 } else if constexpr(std::is_same_v<first_type, entity_type>) {
339 other.emplace(map(pair.first), std::move(pair.second));
340 } else {
341 static_assert(std::is_same_v<second_type, entity_type>, "Neither the key nor the value are of entity type");
342 other.emplace(std::move(pair.first), map(pair.second));
343 }
344 }
345
346 using std::swap;
347 swap(container, other);
348 }

384 void assign(Archive &archive, [[maybe_unused]] Member Other::*...member) {
385 typename entity_traits::entity_type length{};
386 entity_type entt;
387
388 archive(length);
389
390 if constexpr(ignore_as_empty_v<Component>) {
391 while(length--) {
392 archive(entt);
393 restore(entt);
394 reg->template emplace_or_replace<Component>(map(entt));
395 }
396 } else {
397 Component instance;
398
399 while(length--) {
400 archive(entt, instance);
401 (update(instance, member), ...);
402 restore(entt);

404 }
