227 void tick(const Delta delta, void *data = nullptr) {
228 switch(current) {
229 case state::uninitialized:
230 next(std::integral_constant<state, state::uninitialized>{});
231 current = state::running;
232 break;
233 case state::running:
234 next(std::integral_constant<state, state::running>{}, delta, data);
235 break;
236 default:
237 // suppress warnings
238 break;
239 }
240
241 // if it's dead, it must be notified and removed immediately
242 switch(current) {
243 case state::succeeded:
244 next(std::integral_constant<state, state::succeeded>{});
245 current = state::finished;
246 break;
247 case state::failed:
248 next(std::integral_constant<state, state::failed>{});
249 current = state::rejected;
250 break;

252 next(std::integral_constant<state, state::aborted>{});
253 current = state::rejected;
254 break;
255 default:

258 }
