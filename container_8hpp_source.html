70 [[nodiscard]] static iterator insert_or_erase([[maybe_unused]] const meta_ctx &ctx, [[maybe_unused]] any &container, [[maybe_unused]] const any &handle, [[maybe_unused]] meta_any &value) {
71 if constexpr(is_dynamic_sequence_container<Type>::value) {
72 if(auto *const cont = any_cast<Type>(&container); cont) {
73 typename Type::const_iterator it{};
74
75 if(auto *non_const = any_cast<typename Type::iterator>(&handle); non_const) {
76 it = *non_const;
77 } else {
78 it = any_cast<const typename Type::const_iterator &>(handle);
79 }
80
81 if(value) {
82 // this abomination is necessary because only on macos value_type and const_reference are different types for std::vector<bool>
83 if(value.allow_cast<typename Type::const_reference>() || value.allow_cast<typename Type::value_type>()) {
84 const auto *element = value.try_cast<std::remove_reference_t<typename Type::const_reference>>();
85 return iterator{ctx, cont->insert(it, element ? *element : value.cast<typename Type::value_type>())};
86 }
87 } else {
88 return iterator{ctx, cont->erase(it)};
89 }
90 }
91 }

120 }

126 [[nodiscard]] static size_type insert_or_erase(any &container, meta_any &key, meta_any &value) {
127 if(auto *const cont = any_cast<Type>(&container); cont && key.allow_cast<const typename Type::key_type &>()) {
128 if(value) {
129 if constexpr(key_only) {
130 return cont->insert(key.cast<const typename Type::key_type &>()).second;
131 } else {
132 return value.allow_cast<const typename Type::mapped_type &>() && cont->emplace(key.cast<const typename Type::key_type &>(), value.cast<const typename Type::mapped_type &>()).second;
133 }
134 } else {
135 return cont->erase(key.cast<const typename Type::key_type &>());
136 }
137 }
138
139 return 0u;
140 }
141
142 [[nodiscard]] static iterator find(const meta_ctx &ctx, any &container, meta_any &key) {
143 if(key.allow_cast<const typename Type::key_type &>()) {
144 if(auto *const cont = any_cast<Type>(&container); cont) {
145 return iterator{ctx, std::bool_constant<key_only>{}, cont->find(key.cast<const typename Type::key_type &>())};
146 }
147
148 return iterator{ctx, std::bool_constant<key_only>{}, any_cast<const Type &>(container).find(key.cast<const typename Type::key_type &>())};
149 }

151 return iterator{};
152 }
